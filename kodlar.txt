function pidUpdate(ctrl,err,dt){
  ctrl.integral+=err*dt; ctrl.integral=Math.max(-20,Math.min(20,ctrl.integral));
  const d=(err-ctrl.prevErr)/dt; ctrl.prevErr=err;
  return ctrl.kp*err+ctrl.ki*ctrl.integral+ctrl.kd*d;
}


function clamp(v,mn,mx){return Math.max(mn,Math.min(mx,v))}
function rnd(v,d=1){return parseFloat(v.toFixed(d))}
function noise(a){return (Math.random()-0.5)*a*2}

function drawAH(){
  const c=document.getElementById('ah-canvas');
  const ctx=c.getContext('2d'); const w=200,h=120;
  ctx.clearRect(0,0,w,h);
  ctx.save(); ctx.translate(w/2,h/2); ctx.rotate(-S.bank*Math.PI/180);
  const off=S.pitch*2.5;
  // sky
  ctx.fillStyle='#1B4F8A'; ctx.fillRect(-200,-200,400,200+off); 
  // ground
  ctx.fillStyle='#6B4226'; ctx.fillRect(-200,off,400,200);

  ctx.strokeStyle='#fff'; ctx.lineWidth=2; ctx.beginPath(); ctx.moveTo(-200,off); ctx.lineTo(200,off); ctx.stroke();

    if(p===0)continue; const y=off-p*2.5;
    const len=p%10===0?30:18;
    ctx.strokeStyle='rgba(255,255,255,0.7)'; ctx.lineWidth=1;
    ctx.beginPath(); ctx.moveTo(-len/2,y); ctx.lineTo(len/2,y); ctx.stroke();

  }
  ctx.restore();

  ctx.strokeStyle='#FFD700'; ctx.lineWidth=2;
  ctx.beginPath(); ctx.moveTo(w/2-30,h/2); ctx.lineTo(w/2-10,h/2); ctx.moveTo(w/2+10,h/2); ctx.lineTo(w/2+30,h/2); ctx.moveTo(w/2,h/2); ctx.lineTo(w/2,h/2-8); ctx.stroke();

// === PID LOOP ===
function runPID(dt){
  const pitchErr=S.targetPitch-S.pitch;
  const rollErr=S.targetBank-S.bank;
  const hdgErr=((S.targetHdg-S.hdg+540)%360)-180;
  S.elevCmd=clamp(pidUpdate(S.pidPitch,pitchErr,dt)*15,-25,25);
  S.ailCmd=clamp(pidUpdate(S.pidRoll,rollErr,dt)*10,-30,30);
  S.rudCmd=clamp(pidUpdate(S.pidYaw,hdgErr*0.2,dt)*5,-20,20);
  S.pitch+=pitchErr*0.02; S.bank+=rollErr*0.02; S.hdg=((S.hdg+hdgErr*0.005)+360)%360;
}

// === MAIN SIMULATION LOOP ===
function update(){
  tick++; const dt=0.05;
  S.simTime+=dt;
  // apply scenario
  scenarios[S.mode].apply();
  // PID control
  runPID(dt);
  // physics drift
  S.ias=clamp(S.ias+noise(0.3)+pidNoise(),-0,600);
  S.alt+=S.vs*dt/60+noise(S.mode==='gpws'?-15:2);
  S.fuel=Math.max(0,S.fuel-0.8);
  S.lat=rnd(S.lat+0.0001*(Math.sin(S.hdg*Math.PI/180)),4);
  S.lon=rnd(S.lon+0.0001*(Math.cos(S.hdg*Math.PI/180)),4);
  S.tas=rnd(S.ias*1.115,0);
  S.mach=rnd(S.ias/350,2);
  S.eta=Math.max(0,S.eta-dt/60);
  if(S.mode==='normal'){S.aoa=rnd(3.2+noise(0.2),1);S.g=rnd(1.0+noise(0.05),2);}
  // update DOM
  document.getElementById('v-ias').textContent=Math.round(S.ias);
  document.getElementById('v-tas').textContent=Math.round(S.tas);
  document.getElementById('v-mach').textContent=S.mach;
  document.getElementById('v-vs').textContent=Math.round(S.vs);
  document.getElementById('v-pitch').textContent=rnd(S.pitch);
  document.getElementById('v-bank').textContent=rnd(S.bank);
  document.getElementById('v-alt').textContent=Math.round(S.alt).toLocaleString();
  document.getElementById('v-hdg').textContent=Math.round(S.hdg);
  document.getElementById('v-pos').textContent=`${S.lat}°N ${S.lon}°E`;
  document.getElementById('v-eta').textContent=`${String(Math.floor(S.eta/60)).padStart(2,'0')}:${String(Math.round(S.eta%60)).padStart(2,'0')}`;
  document.getElementById('g-val').textContent=rnd(S.g)+'g';
  document.getElementById('g-fill').style.width=clamp(S.g/3*100,0,100)+'%';
  document.getElementById('aoa-val').textContent=rnd(S.aoa)+'°';
  document.getElementById('aoa-fill').style.width=clamp(S.aoa/25*100,0,100)+'%';
  setEngineBar('e1',S.e1); setEngineBar('e2',S.e2);
  document.getElementById('egt-val').textContent=Math.round(S.egt)+'°';
  document.getElementById('egt-fill').style.width=clamp(S.egt/900*100,0,100)+'%';
  document.getElementById('egt-fill').className='gauge-fill '+(S.egt>850?'fill-red':S.egt>750?'fill-amber':'fill-amber');
  document.getElementById('fuel-val').textContent=Math.round(S.fuel).toLocaleString();
  document.getElementById('fuel-fill').style.width=clamp(S.fuel/20000*100,0,100)+'%';
  document.getElementById('fuel-fill').className='gauge-fill '+(S.fuel<3000?'fill-red':S.fuel<6000?'fill-amber':'fill-blue');
  // PID outputs
  setPIDBar('pid-elev',S.elevCmd,25); setPIDBar('pid-ail',S.ailCmd,30); setPIDBar('pid-rud',S.rudCmd,20);
  document.getElementById('pid-elev-v').textContent=rnd(S.elevCmd)+'°';
  document.getElementById('pid-ail-v').textContent=rnd(S.ailCmd)+'°';
  document.getElementById('pid-rud-v').textContent=rnd(S.rudCmd)+'°';
  // speed tape
  document.getElementById('spd-ptr').textContent=Math.round(S.ias);
  document.getElementById('spd-ptr').style.top=clamp(50-(S.ias-260)*0.3,5,70)+'px';
  // scenario label
  document.getElementById('scenario-lbl').textContent=scenarios[S.mode].label;
  document.getElementById('fcs-mode-lbl').textContent=S.fcsLaw;
  document.getElementById('phase-val').textContent=S.phase;
  // UTC
  const now=new Date();
  document.getElementById('utc-val').textContent=now.toUTCString().slice(17,25);
  // danger coloring
  document.getElementById('v-ias').className='metric-value'+(S.ias<200||S.ias>380?' crit':S.ias<230||S.ias>340?' warn':'');
  document.getElementById('aoa-val').className='metric-value'+(S.aoa>15?' crit':S.aoa>10?' warn':'ok');
  // auto alerts (throttle)
  if(tick%120===0&&S.mode==='normal') addAlert('info','FCS nagel check — nominal');
  if(S.fuel<3000&&tick%60===0) addAlert('warn','FUEL LOW — '+Math.round(S.fuel)+'kg');
  if(S.aoa>15&&tick%30===0) addAlert('crit','STALL — AoA krit '+rnd(S.aoa)+'°');
  if(S.egt>850&&tick%60===0) addAlert('crit','EGT yüksek — '+Math.round(S.egt)+'°C');
  drawAH(); drawMap();
}
